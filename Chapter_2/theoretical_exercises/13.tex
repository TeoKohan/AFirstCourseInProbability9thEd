\item Prove that $P(EF^c) = P(E) - P(EF)$.

\begin{proof}[Demostraci\'on] de la equivalencia anterior.
    \begin{align*}
        &\phantom{{}={}} P(E)\\
        &= P(E(F\cup F^c))\\
        &= P((EF)\cup (EF^c))\\
        &= P(EF) + P(EF^c) - P(EFEF^c)\\
        &= P(EF) + P(EF^c) \qedhere
    \end{align*}
\end{proof}
