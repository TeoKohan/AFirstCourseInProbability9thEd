\item Suppose that $A$ and $B$ are mutually exclusive events for which $P(A) = .3$ and $P(B) = .5$. What is the probability that
\begin{enumerate}
    \item either $A$ or $B$ occurs?
    \begin{align*}
        &\phantom{\;=\;} P(A\cup B)\\
        &= P(A) + P(B)\\
        &= .3 + .5\\
        &= .8
    \end{align*}
    \item $A$ occurs but $B$ does not?
    \begin{align*}
        &\phantom{\;=\;} P(AB^c)\\
        &= P(A) - P(AB)\\
        &= .3 - .5\\
        &= .3
    \end{align*}
    \item both $A$ and $B$ occur?
    \begin{align*}
        P(AB) &= 0
    \end{align*}
\end{enumerate}
